62static inline void pack_ipmi_resp(struct ipmi_response *resp, ipmi_msg *ipmi_resp)
63{
64 resp->netfn = (ipmi_resp->netfn + 1) << 2; // ipmi netfn response package
65 resp->cmd = ipmi_resp->cmd;
66 resp->cmplt_code = ipmi_resp->completion_code;
67 if (ipmi_resp->data_len != 0) {
68 memcpy(resp->data, ipmi_resp->data, ipmi_resp->data_len);
69 }
70}
