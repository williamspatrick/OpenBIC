161static inline int calculate_accurate_MBR(uint8_t sensor_num, int val)
162{ // for better accuracy, enlarge SDR to two byte scale
163 if (SDR_M(sensor_num) == 0) {
164 return ((val << 8) * SDR_Rexp(sensor_num));
165 }
166 return ((val << 8) / SDR_M(sensor_num) * SDR_Rexp(sensor_num));
167}
168
169static inline int calculate_MBR(uint8_t sensor_num, int val)
170{
171 if (SDR_M(sensor_num) == 0) {
172 return (val * SDR_Rexp(sensor_num) + round_add(sensor_num, val));
173 }
174 return (val * SDR_Rexp(sensor_num) / SDR_M(sensor_num) + round_add(sensor_num, val));
175}
176
177static inline float convert_MBR_to_reading(uint8_t sensor_num, uint8_t val)
178{
179 if (SDR_M(sensor_num) == 0) {
180 return (val - round_add(sensor_num, val)) / SDR_Rexp(sensor_num);
181 }
182
183 return (val - round_add(sensor_num, val)) * SDR_M(sensor_num) / SDR_Rexp(sensor_num);
184}
